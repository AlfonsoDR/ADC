24 // the alternate clock is connected to OSCERCLK (16 MHz).
25 // datasheet says ADC clock should be 2 to 12 MHz for 16 bit mode
26 // datasheet says ADC clock should be 1 to 18 MHz for 8-12 bit mode
27 // calibration works best when averages are 32 and speed is less than 4 MHz
28 #if F_BUS == 48000000
29  #define ADC_CFG1_3MHZ ADC_CFG1_ADIV(3) + ADC_CFG1_ADICLK(1) // Clock divide select: 3=div8 + Input clock: 1=bus/2
30  #define ADC_CFG1_6MHZ ADC_CFG1_ADIV(2) + ADC_CFG1_ADICLK(1) // Clock divide select: 2=div4 + Input clock: 1=bus/2
31  #define ADC_CFG1_12MHZ ADC_CFG1_ADIV(1) + ADC_CFG1_ADICLK(1) // Clock divide select: 1=div2 Input clock: 1=bus/2
32  #define ADC_CFG1_24MHZ ADC_CFG1_ADIV(0) + ADC_CFG1_ADICLK(1) // Clock divide select: 0=direct + Input clock: 1=bus/2
33 #elif F_BUS == 24000000
